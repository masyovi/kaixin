    22→function Reveal({
    23→  children,
    24→  delay = 0,
    25→  className = "",
    26→  y = 20,
    27→}: {
    28→  children: React.ReactNode;
    29→  delay?: number;
    30→  className?: string;
    31→  y?: number;
    32→}) {
    33→  const ref = useRef(null);
    34→  const visible = useInView(ref, { once: true, margin: "-40px" });
    35→  return (
    36→    <motion.div
    37→      ref={ref}
    38→      initial={{ opacity: 0, y }}
    39→      animate={visible ? { opacity: 1, y: 0 } : {}}
    40→      transition={{
    41→        duration: 0.5,
    42→        delay,
    43→        ease: [0.22, 1, 0.36, 1],
    44→      }}
    45→      className={className}
    46→    >
    47→      {children}
    48→    </motion.div>
    49→  );
    50→}

    80→function CountUp({ target, suffix = "" }: { target: number; suffix?: string }) {
    81→  const [n, setN] = useState(0);
    82→  const ref = useRef(null);
    83→  const visible = useInView(ref, { once: true });
    84→  useEffect(() => {
    85→    if (!visible) return;
    86→    let c = 0;
    87→    const step = target / 50;
    88→    const t = setInterval(() => {
    89→      c += step;
    90→      if (c >= target) { setN(target); clearInterval(t); }
    91→      else setN(Math.floor(c));
    92→    }, 25);
    93→    return () => clearInterval(t);
    94→  }, [visible, target]);
    95→  return <span ref={ref}>{n}{suffix}</span>;
    96→}
