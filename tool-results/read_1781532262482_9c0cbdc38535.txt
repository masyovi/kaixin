    24→
    25→function In({ children, delay = 0, className = "" }: { children: React.ReactNode; delay?: number; className?: string }) {
    26→  const r = useRef(null);
    27→  const v = useInView(r, { once: true, margin: "-50px" });
    28→  return (
    29→    <motion.div ref={r} initial={{ opacity: 0, y: 24 }} animate={v ? { opacity: 1, y: 0 } : {}} transition={{ duration: 0.5, delay, ease: "easeOut" }} className={className}>
    30→      {children}
    31→    </motion.div>
    32→  );
    33→}
    34→
    35→function Count({ target, s = "" }: { target: number; s?: string }) {
    36→  const [n, setN] = useState(0);
    37→  const r = useRef(null);
    38→  const v = useInView(r, { once: true });
    39→  useEffect(() => {
    40→    if (!v) return;
    41→    let c = 0;
    42→    const step = target / 50;
    43→    const t = setInterval(() => { c += step; if (c >= target) { setN(target); clearInterval(t); } else setN(Math.floor(c)); }, 30);
    44→    return () => clearInterval(t);
    45→  }, [v, target]);
    46→  return <span ref={r}>{n.toLocaleString()}{s}</span>;
    47→}

    53→    </span>
